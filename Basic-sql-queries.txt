/*Create an instructor table Create the INSTRUCTOR table as defined above. Have the ins_id be the primary key, and ensure the lastname and firstname are not null.
(Hint: ins_id is of type INTEGER, country of type CHAR(2), and rest of the fields VARCHAR)*/


create table INSTRUCTOR (
	Ins_Id int PRIMARY KEY NOT NULL,
	lastname VARCHAR(15) NOT NULL,
	firstname VARCHAR(15) NOT NULL,
	city VARCHAR(15),
	country CHAR(2)
	
	);
